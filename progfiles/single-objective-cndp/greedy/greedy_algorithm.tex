\begin{algorithm}[t]
  \caption{iGreedy}\label{alg:GREEDY_ALGORITMUS}
  \begin{algorithmic}[1]
    \Require $G, k$
    \State $S \leftarrow \Call{Vertex Cover}{G}$

    \While{$\abs{S} > k$}
    \State $B \leftarrow \argmin_{i \in S} f(S \setminus \left\{ i \right\})$
    \State $S \leftarrow S \setminus \left\{ \Call{Select}{B} \right\}$
    \EndWhile

    \State \Return $S$
  \end{algorithmic}
\end{algorithm}
