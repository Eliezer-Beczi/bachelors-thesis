\begin{algorithm}[h]
  \caption{Random Solution}\label{alg:GENETIKUS-ALGORITMUS:RANDOM-SOLUTION}
  \begin{algorithmic}[1]
    \Require $G, k$
    \State $S \leftarrow V$

    \While{$\abs{S} > k$}
    \State $elem \leftarrow \Call{Select}{S}$
    \State $S \leftarrow S \setminus \left\{ elem \right\}$
    \EndWhile

    \State \Return $S$
  \end{algorithmic}
\end{algorithm}
